



# 闭包

## 作用域

```js
var a = 1;
function f1() {
  var a = 2;
  function f2() {
    var a = 3;
    console.log(a);//3
  }
  return f2(); //原博代码没有return f2()，执行函数时输出undefined
}
```



```js
function f1() {
  var a = 2;
  function f2() {
    console.log(a);//2
  }
  return f2;
}

var x = f1();
x();
```

不用返回函数，给 f3 赋值。先执行 f1()，再执行 f3()。

```js
var f3;

function f1() {
    var a = 2;
    f3 = function() {
        console.log(a);
    }
}

f1(); // undefined
f3(); // 2
```
